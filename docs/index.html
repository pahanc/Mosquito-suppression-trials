Simulating mosquito counts from the posterior of the fitted geostatistical model

load("poiss_PSC.res") # load the INLA fitted model object
load("metadata.RData") # load the metadata
no_samp<-60 # Set the number of posterior samples to draw
inla_mod_samps<-inla.posterior.sample(no_samp,poiss.res)# Draw samples from the fitted model

Simulate mosquito counts using the fitted random effects

1. Get posterior draws of the spatially correlated random effects

#Make the INLA projector matrix
A.pred<-inla.spde.make.A(metadata$BMBS_mesh,loc=metadata$long_lat,group=metadata$samp_months,group.mesh=metadata$mesh1d)

#Get the indices of the posterior draws of the random effects of the spatially correlated
#Gaussian Markov Random Field f() at the mesh node locations
ids<-lapply('BMBS.field',function(x) grep(x,rownames(inla_mod_samps[[1]]$latent),fixed=TRUE))

#Get the values of the posterior draws of the random effects of the spatially correlated
#Gaussian Markov Random Field f() at the mesh node locations
pred.field.f<-function(s) s$latent[ids[[1]]]
pred.field<-sapply(inla_mod_samps,pred.field.f)

#Project the values of the posterior draw of the GMRF across the locations of the sampled compounds
lp_i<-A.pred%*%pred.field

2. Add compound-level noise

ids2<-lapply('Precision',function(x) grep(x,names(inla_mod_samps[[1]]$hyperpar)))
obs.noise.f<-function(s) s$hyperpar[unlist(ids2)[1]] 
obs.noise.site<-sapply(inla_mod_samps,obs.noise.f)#Get the value of the posterior of the precision of the compound-level random effect
obs.noise.site.sim<-rnorm(n=nrow(lp_i)*no_samp,mean=0,sd=1/sqrt(obs.noise.site[1]))
#Draw samples of the random effect for each compound location for each posterior draw from the fitted model
obs.noise.site.sim<-matrix(obs.noise.site.sim,nrow=nrow(lp_i),ncol=no_samp)
lp2_i<-lp_i+obs.noise.site.sim

3. Add village-level noise

ids3<-lapply('Precision',function(x) grep(x,names(inla_mod_samps[[1]]$hyperpar)))
obs.noise.f<-function(s) s$hyperpar[unlist(ids3)[2]]
obs.noise.village<-sapply(inla_mod_samps,obs.noise.f)
obs.noise.village.sim<-rnorm(n=nrow(lp_i)*no_samp,mean=0,sd=1/sqrt(obs.noise.village[1]))
obs.noise.village.sim<-matrix(obs.noise.village.sim,nrow=nrow(lp_i),ncol=no_samp)
lp3_i<-lp2_i+obs.noise.village.sim

#Simulate Poisson distributed mosquito counts from the linear predictor
for (i in 1:ncol(lp3_i)){
  lp3_i[,i]<-as.numeric(lp3_i[,i])
  for (j in 1:nrow(lp3_i)){
    lp3_i[j,i]<-rpois(1,exp(lp3_i[j,i]))
  }
}

Make the simulated CRCT data sets

We can make the simulated CRCT data sets using the make_sim_counts() function in the following source files:

Make sim data sets parallel.R
Make sim data sets step-wedge.R

For example, to simulate data for a parallel CRCT with 30 clusters per arm with a suppression effect of G=50%:

source("Make sim data sets parallel.R")
counts_df<-make_sim_counts(cluster_per_group=30,G=0.5,runno=1)
